import pandas as pd
import matplotlib.pyplot as plt

# 🔹 Step 1: SHAP (Heart Risk)
shap_df = pd.DataFrame({
    "Feature": model.named_steps["preprocess"].get_feature_names_out(),
    "SHAP Value": shap_values.values[0]
})
shap_df["Feature"] = shap_df["Feature"].str.replace("num__", "").str.replace("cat_", "").str.replace("_", " ").str.title()
shap_df["Abs_Heart_Impact"] = shap_df["SHAP Value"].abs()

# 🔹 Step 2: Simulate Health Score Weights (or extract real ones)
# Example weights (replace with actual health score logic if you have it)
health_weights = {
    "Bmi": 0.3,
    "Systolic Bp": 0.25,
    "Heart Rate": 0.2,
    "Diabetes": 0.15,
    "Smoking Status": 0.1
}
health_df = pd.DataFrame(list(health_weights.items()), columns=["Feature", "Health_Impact"])

# 🔹 Step 3: Merge both impacts
combined = pd.merge(shap_df, health_df, on="Feature", how="outer").fillna(0)
combined["Total_Impact"] = combined["Abs_Heart_Impact"] + combined["Health_Impact"]

# 🔹 Step 4: Top contributors + Others
top_n = 5
top_combined = combined.nlargest(top_n, "Total_Impact")
others_sum = combined["Total_Impact"].sum() - top_combined["Total_Impact"].sum()
others_df = pd.DataFrame([{"Feature": "Other Factors", "Total_Impact": others_sum}])
final_df = pd.concat([top_combined[["Feature", "Total_Impact"]], others_df], ignore_index=True)

# 🔹 Step 5: Friendly Pie Chart
colors = ["#A2D9CE", "#F7DC6F", "#F5B7B1", "#85C1E9", "#D7BDE2", "#D5DBDB"]
plt.figure(figsize=(7, 7))
plt.pie(final_df["Total_Impact"], labels=final_df["Feature"],
        autopct='%1.1f%%', startangle=140, colors=colors,
        wedgeprops={'edgecolor': 'white'})
plt.title("🩺 Key Health Factors Affecting Your Wellness", fontsize=14)
plt.tight_layout()
plt.show()
